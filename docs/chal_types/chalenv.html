 19class ChallengeEnvironment(GeneratedChallenge):
 20    def __init__(self):
 21        self.chal_path_lookup = {}
 22        self.challenge_types = None
 23
 24    def gen_chals(self, env_path):
 25        self._chal_lookup = {}
 26        if self.config is None:
 27            return
 28
 29
 30        tree = {
 31            "name": self.name,
 32            "chal": self,
 33            "children": []
 34        }
 35        if 'chals' not in self.config or self.config['chals'] is None:
 36            return tree
 37        for chal in self.config['chals']:
 38            chal_path = self.chal_path_lookup.get(chal)
 39            if not chal_path or not os.path.isdir(chal_path):
 40                raise Exception(f'challenge directory not found! {chal}')
 41
 42            chal_config = os.path.join(chal_path, 'chal.yaml')
 43            if not os.path.exists(chal_config):
 44                raise Exception(f'could not find challenge config! {chal}')
 45
 46            chal_gen = load_chal_from_config(self.challenge_types, chal_config)
 47
 48            if ChallengeEnvironment in type(chal_gen).__bases__:
 49                chal_gen.chal_host = self.chal_host
 50                chal_gen.chal_path_lookup = self.chal_path_lookup
 51                chal_gen.challenge_types = self.challenge_types
 52                chal_children = chal_gen.gen_chals(chal_path)
 53                chal_gen.do_gen(chal_path)
 54
 55                tree["children"].append(chal_children)
 56            else:
 57                chal_gen.do_gen(chal_path)
 58
 59                tree["children"].append({
 60                    "name": chal_gen.name,
 61                    "chal": chal_gen,
 62                    "children": []
 63                })
 64
 65            self._chal_lookup[chal_gen.name] = chal_gen
 66        return tree

465            # check if the field is just a number, if so skip
466            try:
467                int(entity[key], 10)

475        def force_bytes(s, encoding='utf-8', strings_only=False, errors='strict'):
476            """
477            Similar to smart_bytes, except that lazy instances are resolved to
478            strings, rather than kept as lazy objects.
479            If strings_only is True, don't convert (some) non-string-like objects.
480            """
481            # Handle the common case first for performance reasons.
482            if isinstance(s, bytes):
483                if encoding == 'utf-8':
484                    return s
485                else:
486                    return s.decode('utf-8', errors).encode(encoding, errors)
487            if isinstance(s, memoryview):
488                return bytes(s)
489            return str(s).encode(encoding, errors)
490
491        def pbkdf2(password, salt, iterations, dklen=0, digest=None):
492            """Return the hash of password using pbkdf2."""
493            if digest is None:
494                digest = hashlib.sha256
495            dklen = dklen or None
496            password = force_bytes(password)
497            salt = force_bytes(salt)
498            return hashlib.pbkdf2_hmac(digest().name, password, salt, iterations, dklen)
499
500        def get_pwd_hash(password):
501            salt = 'A'*9
502            iterations = 12000
503            hash = pbkdf2(password, salt, iterations, digest=hashlib.sha256)
504            hash = base64.b64encode(hash).decode('ascii').strip()
505            return "%s$%d$%s$%s" % ("pbkdf2_sha256", iterations, salt, hash)
506

521
522                if 'likers' in entity:

525
526                if 'chal' in entity:

528
529                if (entity_type == 'posts' or entity_type == 'comments') and 'date_added' not in entity:

531
532                out.append({
533                    'fields': entity,
534                    'model': entity_to_model[entity_type],
535                    'pk': n
536                })
