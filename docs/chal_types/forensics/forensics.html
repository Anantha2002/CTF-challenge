575def get4(srcpages):
576    scale = 0.5
577    srcpages = PageMerge() + srcpages
578    x_increment, y_increment = (scale * i for i in srcpages.xobj_box[2:])
579    for i, page in enumerate(srcpages):
580        page.scale(scale)
581        page.x = x_increment if i & 1 else 0
582        page.y = 0 if i & 2 else y_increment
583    return srcpages.render()
